Module type Gfx.Api

module type Api = sig .. end
Main graphics drawing API.

Most functions take a bounds parameter which is a rectangle to which drawing is clipped and also serves as an origin for any coordinates.

type ctx
drawing context
type style
underlying style type
val get_bounds : ctx -> Gfx.rect
get context size
val get_style : Gfx.Style.t -> style
convert our style info to underlying style
val clear : ctx -> unit
clear display
val fill : ctx:ctx -> style:style -> bounds:Gfx.rect -> char -> unit
fill bounds with char given style
val draw_int : ctx:ctx ->
       style:style -> bounds:Gfx.rect -> r:int -> c:int -> int -> unit
draw int (representing unicode value)
val draw_piece : ctx:ctx ->
       style:style -> bounds:Gfx.rect -> r:int -> c:int -> Gfx.piece -> unit
draw piece
val draw_char : ctx:ctx ->
       style:style -> bounds:Gfx.rect -> r:int -> c:int -> char -> unit
draw char
val draw_string : ctx:ctx ->
       style:style -> bounds:Gfx.rect -> r:int -> c:int -> string -> unit
draw string (nothing fancy - horizontal, no breaks)
val draw_box : ctx:ctx -> style:style -> bounds:Gfx.rect -> string -> unit
draw box outline with label
val get : ctx:ctx -> bounds:Gfx.rect -> r:int -> c:int -> int * Gfx.Style.t
get value and style at point
val inv : ctx:ctx -> bounds:Gfx.rect -> r:int -> c:int -> unit
invert fg and bg at point
val bold : ctx:ctx -> bounds:Gfx.rect -> r:int -> c:int -> unit
set bold on point
